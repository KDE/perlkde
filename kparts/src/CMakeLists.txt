include_directories(
    ${CMAKE_SOURCE_DIR}/smoke
    ${CMAKE_SOURCE_DIR}/perl/qtcore/src
    ${PERL_INCLUDE_PATH}
    ${QT_INCLUDES} )

# Run doxsubpp.pl to run xsubpp on KParts.xs
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/KParts.c
                   COMMAND ${PERL_EXECUTABLE} ARGS ${CMAKE_SOURCE_DIR}/perl/qtcore/src/doxsubpp.pl ${PERL_EXECUTABLE} KParts.xs ${CMAKE_CURRENT_BINARY_DIR}/KParts.c
                   DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/KParts.xs
                   WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
)
# Make sure we build KParts.c as c++ code
set_source_files_properties( ${CMAKE_CURRENT_BINARY_DIR}/KParts.c 
    PROPERTIES LANGUAGE CXX
)

set( CMAKE_CXX_FLAGS ${PERL_CXX_FLAGS} )

set(kparts_LIBRARY_SRC
    ${CMAKE_CURRENT_BINARY_DIR}/KParts.c
    kpartshandlers.cpp
)

set(LIBRARY_OUTPUT_PATH ${CMAKE_BINARY_DIR}/perl/blib/arch/auto/KParts)
add_library(perl_kparts MODULE ${kparts_LIBRARY_SRC})

target_link_libraries(perl_kparts
    smokekparts
    perlqtcore4)

set_target_properties(perl_kparts PROPERTIES OUTPUT_NAME "KParts")
set_target_properties(perl_kparts PROPERTIES PREFIX "")

install(TARGETS perl_kparts DESTINATION ${CUSTOM_PERL_SITE_ARCH_DIR}/auto/KParts/)
